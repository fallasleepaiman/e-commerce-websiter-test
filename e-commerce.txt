        const imageGrid = document.getElementById('image-grid');

        // Function to render the correct view based on the URL hash
        function renderView() {
            const productId = window.location.hash.slice(1).split('-')[1];
            const product = products.find(p => p.id === productId);

            if (product) {
                // Show product details view and hide main page sections

                productName.textContent = product.name;
                productPrice.textContent = product.price;
                productDescription.textContent = product.description;
                
                productListSection.classList.add('hidden');
                heroSection.classList.add('hidden');
                imageGrid.classList.add('hidden');
                productDetailsSection.classList.remove('hidden');
            } else {

                productListSection.classList.remove('hidden');
                heroSection.classList.remove('hidden');
                imageGrid.classList.remove('hidden');
            }
        }

        // Listen for changes to the URL hash
        window.addEventListener('hashchange', renderView);

        // Initial render on page load
